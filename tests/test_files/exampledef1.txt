DEVICE  G1: AND 2 INPUTS, 
        G2: AND 2 INPUTS, 
        G3: NOR 2 INPUTS,
        G4: NOR 2 INPUTS,
        SW1: SWITCH 0, / R /
        SW2: SWITCH 1, / S / 
        CLK1: CLOCK 10;

CONNECT SW1 -> G1.I1,
        SW2 -> G2.I2,
        CLK1 -> G1.I2,
        CLK1 -> G2.I1,
        G1 -> G3.I1,
        G2 -> G4.I2,
        G3 -> G4.I1,
        G4 -> G3.I2;

MONITOR G3, / Q(t) /
        G4; / Q(t)' /
